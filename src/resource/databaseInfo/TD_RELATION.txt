-- Create table
create table TD_RELATION
(
  id          VARCHAR2(32) not null,
  elementname VARCHAR2(200),
  fieldname   VARCHAR2(100),
  tablename   VARCHAR2(100),
  type        VARCHAR2(50),
  parten      VARCHAR2(200)
);
-- Add comments to the table 
comment on table TD_RELATION
  is '元素和结构数据库存放表';
-- Add comments to the columns 
comment on column TD_RELATION.elementname
  is '分类标准元素的名称';
comment on column TD_RELATION.fieldname
  is '对应的数据库表字段名称';
comment on column TD_RELATION.tablename
  is '对应的数据库表名称';
comment on column TD_RELATION.type
  is '存放数据库的类型';
comment on column TD_RELATION.parten
  is '数据库的区块';
-- Create/Recreate primary, unique and foreign key constraints 
alter table TD_RELATION
  add constraint TD_RELATION_PK primary key (ID)
  using index;
